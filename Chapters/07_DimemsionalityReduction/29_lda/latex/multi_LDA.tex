\newcommand\setrowY[9]{
  \setcounter{col}{0}
  \foreach \n [evaluate={\s=\n<1 ? "C0" : (\n<2 ? "C1" : (\n<3 ? "C2": (\n < 4? "C3" : "C4")));}] in {#1, #2, #3, #4, #5, #6, #7, #8, #9} {
    \edef\x{\value{col}}
    \edef\y{\value{row}+1}    
    \draw [\s, draw = gray!30] (\x,\y) rectangle +(1,-1);
    \stepcounter{col}
  }
  \stepcounter{row}
}
