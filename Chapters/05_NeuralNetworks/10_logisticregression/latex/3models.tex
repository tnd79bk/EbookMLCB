    % Draw the input layer nodes

        \node[input neuron] (I-\name) at (0,-\y) {};


    % Draw the output layer nodes
    \foreach \name / \y in {3}
        \path[yshift=0.5cm]
            node[linear neuron ] (O-\name) at (\layersep,-\y cm) {};
    % hidden layer.
    \foreach \source in {1,...,4}
